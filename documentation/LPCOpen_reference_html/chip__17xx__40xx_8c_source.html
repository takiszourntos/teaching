61 /* Sets up USB PLL, all needed clocks and enables USB PHY on the chip. USB pins which are
62  muxed to different pads are not initialized here. This routine assumes that the XTAL
63  OSC is enabled and running prior to this call. */
64 void Chip_USB_Init(void)
65 {
66 
67 #if defined(CHIP_LPC175X_6X)
68  /* Setup USB PLL1 for a 48MHz clock
69  Input clock rate (FIN) is main oscillator = 12MHz
70  PLL1 Output = USBCLK = 48MHz = FIN * MSEL, so MSEL = 4.
71  FCCO = USBCLK = USBCLK * 2 * P. It must be between 156 MHz to 320 MHz.
72  so P = 2 and FCCO = 48MHz * 2 * 2 = 192MHz */
73  Chip_Clock_SetupPLL(SYSCTL_USB_PLL, 3, 1); /* Multiply by 4, Divide by 2 */
74 
75  /* Use PLL1 output as USB Clock Source */
76  /* Enable PLL1 */
77  Chip_Clock_EnablePLL(SYSCTL_USB_PLL, SYSCTL_PLL_ENABLE);
78 
79  /* Wait for PLL1 to lock */
80  while (!Chip_Clock_IsUSBPLLLocked()) {}
81 
82  /* Connect PLL1 */
83  Chip_Clock_EnablePLL(SYSCTL_USB_PLL, SYSCTL_PLL_ENABLE | SYSCTL_PLL_CONNECT);
84 
85  /* Wait for PLL1 to be connected */
86  while (!Chip_Clock_IsUSBPLLConnected()) {}
87 
88 #else
89 
90  /* Select XTAL as clock source for USB block and divider as 1 */
91  LPC_SYSCTL->USBCLKSEL = 0x1;
92  /* Setup USB PLL1 for a 48MHz clock
93  Input clock rate (FIN) is main oscillator = 12MHz
94  PLL output = 48MHz = FIN * MSEL, so MSEL = 4
95  FCCO must be between 156 MHz to 320 MHz, where FCCO = PLL output * 2 * P,
96  so P = 2 and FCCO = 48MHz * 2 * 2 = 192MHz */
97  Chip_Clock_SetupPLL(SYSCTL_USB_PLL, 3, 1);
98 
99  /* Wait for USB PLL to lock */
100  while ((Chip_Clock_GetPLLStatus(SYSCTL_USB_PLL) & SYSCTL_PLLSTS_LOCKED) == 0) {}
101 
102  /* Select PLL1/USBPLL as clock source for USB block and divider as 1 */
103  LPC_SYSCTL->USBCLKSEL = (SYSCTL_USBCLKSRC_USBPLL << 8) | 0x01;
104 
105 #endif /* defined(CHIP_LPC175X_6X) */
106 
107  /* Enable AHB clock to the USB block and USB RAM. */
108  Chip_Clock_EnablePeriphClock(SYSCTL_CLOCK_USB);
109 
110 }
