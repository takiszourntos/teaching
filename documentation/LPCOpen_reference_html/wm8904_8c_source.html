152 /* Read data from UDA register */
153 uint16_t WM8904_REG_Read(uint8_t reg) {
154  uint8_t rx_data[2];
155  if (Chip_I2C_MasterCmdRead(WM8904_I2C_BUS, I2CDEV_WM8904_ADDR, reg, rx_data, 2) == 2) {
156  return (rx_data[0] << 8) | rx_data[1];
157  }
158  return 0;
159 }

161 /* Write data to Codec register */
162 void WM8904_REG_Write(uint8_t reg, uint16_t val)
163 {
164  uint8_t dat[3];
165  dat[0] = reg; dat[1] = val >> 8; dat[2] = val & 0xFF;
166  Chip_I2C_MasterSend(WM8904_I2C_BUS, I2CDEV_WM8904_ADDR, dat, sizeof(dat));
167 }

178 /* Multiple value verification function */
179 int WM8904_REG_VerifyMult(uint8_t reg, const uint8_t *value, uint8_t *buff, int len)
180 {
181  int i;
182  if (Chip_I2C_MasterCmdRead(WM8904_I2C_BUS, I2CDEV_WM8904_ADDR, reg, buff, len) != len) {
183  return 0; /* Partial read */
184 
185  }
186  /* Compare the values */
187  for (i = 0; i < len; i++) {
188  if (value[i] != buff[i]) {
189  break;
190  }
191  }
192 
193  return i == len;

241 /* Write multiple registers in one go */
242 int WM8904_MultiRegWrite(const WM8904_Init_Seq_t* seq, uint32_t cnt)
243 {
244  uint16_t rd_val = 1;
245  uint32_t i;
246 
247  for (i = 0; i < cnt; i++) {
248  /* check if it is delay register */
249  if (seq[i].reg_adr == 0xFF) {
250 
251  delay(seq[i].reg_val * 1000);
252  do {
253  rd_val = WM8904_REG_Read(0x70);
254  } while (rd_val & 1);
255  } else {
256  WM8904_REG_Write(seq[i].reg_adr, seq[i].reg_val);
257  rd_val = WM8904_REG_Read(seq[i].reg_adr);
258  }
259  }
260  return (SUCCESS);
261 }
