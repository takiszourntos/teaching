142  uint16_t EntryNum,
143  bool IsFullCANEntry)
144 {
145  uint16_t i;
146  uint16_t CurID = 0;
147  uint16_t Entry;
148  uint16_t EntryCnt = 0;

155  }

186 /* Shift a number of entries down 1 position */
187 STATIC void shiftSTDEntryDown(uint32_t *arr, int32_t num)
188 {
189  uint32_t i = 0;
190  uint32_t prevRow, curRow;
191 
192  if (num <= 0) {
193  return;
194  }
195 
196  prevRow = arr[0];
197  arr[0] = ((prevRow & 0xFFFF0000) >> 16);
198  for (i = 0; i < (num / 2); i++) {
199  curRow = arr[i + 1];
200  arr[i + 1] = ((prevRow & 0xFFFF) << 16) | ((curRow & 0xFFFF0000) >> 16);
201  prevRow = curRow;
202  }
203  if ((num % 2) == 0) {
204  uint8_t CtrlNo = (prevRow >> CAN_STD_ENTRY_CTRL_NO_POS) & CAN_STD_ENTRY_CTRL_NO_MASK;
205  arr[num / 2] = ((prevRow & 0xFFFF) << 16) | createUnUsedSTDEntry(CtrlNo);
206  }
207 }

209 /* Shift a number of entries up 1 position */
210 STATIC void shiftSTDEntryUp(uint32_t *arr, int32_t num)
211 {
212  int32_t i = 0;
213  uint32_t prevRow = 0, curRow = 0;
214  uint8_t CtrlNo = 0;
215 
216  if (num <= 0) {
217  return;
218  }
219 
220  curRow = arr[((num + 1) / 2) - 1];
221  CtrlNo = (curRow >> CAN_STD_ENTRY_CTRL_NO_POS) & CAN_STD_ENTRY_CTRL_NO_MASK;
222 
223  /* If num is odd, the last row only includes one item. Therefore, there is nothing
224  to do with it. If num is even, shift the last item to one place of the row.*/
225  if ((num % 2) == 0) {
226  arr[((num + 1) / 2) - 1] = ((curRow & 0xFFFF) << 16) | createUnUsedSTDEntry(CtrlNo);
227  }
228  /* Shift from the row before the last row */
229  for (i = ((num + 1) / 2) - 1; i > 0; i--) {
230  prevRow = arr[i - 1];
231  arr[i - 1] = ((curRow & 0xFFFF0000) >> 16) | ((prevRow & 0xFFFF) << 16);
232  curRow = prevRow;
233  }
234 }

282 /* Get entry value from the given start index of the given array. byteNum is 2 or 4 bytes */
283 STATIC uint32_t getArrayVal(uint8_t *arr, uint32_t startIndex, uint8_t byteNum)
284 {
285  uint8_t i = 0;
286  uint32_t retVal = 0;
287  uint32_t index;
288 
289  index = startIndex * byteNum;
290  if (byteNum == 2) { // each entry uses 2 bytes
291  if (startIndex % 2) {
292  index -= 2; // little endian
293  }
294  else {
295  index += 2;
296  }
297  }
298 
299  for (i = 0; i < byteNum; i++) {
300  retVal |= arr[index + i] << (8 * i);
301  }
302  return retVal;

305 /* Search the index to insert the new entry */
306 STATIC int32_t searchInsertIndex(uint32_t *arr, uint32_t arrNum, uint32_t val, uint32_t mask, uint8_t unitSize)
307 {
308  uint32_t LowerIndex, UpperIndex, MidIndex;
309  uint32_t MidVal;
310  uint8_t *byteArray = (uint8_t *) arr;
311 
312  if (arrNum == 0) {
313  return 0; /* Insert into the first line */
314 
315  }
316  LowerIndex = 0;
317  UpperIndex = arrNum - 1;
318  while (LowerIndex + 1 < UpperIndex) {
319  MidIndex = (LowerIndex + UpperIndex) / 2;
320  MidVal = getArrayVal(byteArray, MidIndex, unitSize) & mask;
321  if (MidVal == val) {
322  return -1; /* The new value is already in the array */
323  }
324  else if (MidVal < val) {
325  LowerIndex = MidIndex + 1;
326  }
327  else {
328  UpperIndex = MidIndex - 1;
329  }
330  }
331 
332  if (((getArrayVal(byteArray, LowerIndex, unitSize) & mask) == val) ||
333  ((getArrayVal(byteArray, UpperIndex, unitSize) & mask) == val)) {
334  return -1; /* The new value is already in the array */
335  }
336  if ((getArrayVal(byteArray, LowerIndex, unitSize) & mask) > val) {
337  return LowerIndex;
338  }
339  if ((getArrayVal(byteArray, UpperIndex, unitSize) & mask) < val) {
340  return UpperIndex + 1;
341  }
342 
343  return UpperIndex;
344 }

346 /* Insert an entry into FullCAN Table */
347 STATIC Status insertSTDEntry(LPC_CANAF_T *pCANAF,
348  LPC_CANAF_RAM_T *pCANAFRam,
349  CAN_STD_ID_ENTRY_T *pEntry,
350  bool IsFullCANEntry)
351 {
352  int32_t IDIndex = 0;
353  uint16_t StartRow, EndRow;
354  uint16_t EntryCnt = 0;
355  uint16_t i = 0;
356  uint32_t tmp = 0;
357  CAN_AF_MODE_T CurMode = Chip_CAN_GetAFMode(pCANAF);
358 
359  if (getTotalEntryNum(pCANAF) >= CANAF_RAM_ENTRY_NUM) {
360  return ERROR;
361  }
362 
363  /* Check if a number of entries in section is odd or even */
364  if (IsFullCANEntry) {
365  getSectionAddress(pCANAF, CANAF_RAM_FULLCAN_SEC, &StartRow, &EndRow);
366  }
367  else {
368  getSectionAddress(pCANAF, CANAF_RAM_SFF_SEC, &StartRow, &EndRow);
369 
370  }
371 
372  if (EndRow > StartRow) {
373  EntryCnt = (EndRow - StartRow + 1) * 2;
374  if ((((pCANAFRam->MASK[EndRow] >> CAN_STD_ENTRY_ID_POS) & CAN_STD_ENTRY_ID_MASK) ==
375  0) &&
376  (((pCANAFRam->MASK[EndRow] >>
377  CAN_STD_ENTRY_DISABLE_POS) & CAN_STD_ENTRY_DISABLE_MASK) == 1)) { /* Unsed entry */
378  EntryCnt -= 1;
379  }
380 
381  }
382 
383  /* Search for Index of new entry */
384  IDIndex = searchInsertIndex((uint32_t *) &pCANAFRam->MASK[StartRow],
385  EntryCnt,
386  pEntry->ID_11 & CAN_STD_ENTRY_ID_MASK,
387  CAN_STD_ENTRY_ID_MASK,
388  sizeof(uint16_t));
389  if ((IDIndex == -1) || (IDIndex > EntryCnt )) {
390  return ERROR;
391  }
392 
393  /* AF Off */
394  Chip_CAN_SetAFMode(pCANAF, CAN_AF_OFF_MODE);
395 
396  /* Move all remaining sections one place down
397  if new entry will increase FullCAN list */
398  if ((EntryCnt % 2) == 0) {
399  uint16_t StartAddr, EndAddr;
400 
401  for (i = getTotalEntryNum(pCANAF); i > EndRow; i--) {
402  pCANAFRam->MASK[i] = pCANAFRam->MASK[i - 1];
403  }
404 
405  if (IsFullCANEntry) {
406  getSectionAddress(pCANAF, CANAF_RAM_FULLCAN_SEC, &StartAddr, &EndAddr);
407  setSectionEndAddress(pCANAF, CANAF_RAM_FULLCAN_SEC, EndAddr + 2);
408  }
409  getSectionAddress(pCANAF, CANAF_RAM_SFF_SEC, &StartAddr, &EndAddr);
410  setSectionEndAddress(pCANAF, CANAF_RAM_SFF_SEC, EndAddr + 2);
411  getSectionAddress(pCANAF, CANAF_RAM_SFF_GRP_SEC, &StartAddr, &EndAddr);
412  setSectionEndAddress(pCANAF, CANAF_RAM_SFF_GRP_SEC, EndAddr + 2);
413  getSectionAddress(pCANAF, CANAF_RAM_EFF_SEC, &StartAddr, &EndAddr);
414  setSectionEndAddress(pCANAF, CANAF_RAM_EFF_SEC, EndAddr + 2);
415  getSectionAddress(pCANAF, CANAF_RAM_EFF_GRP_SEC, &StartAddr, &EndAddr);
416  setSectionEndAddress(pCANAF, CANAF_RAM_EFF_GRP_SEC, EndAddr + 2);
417  }
418 
419  /* Shift rows behind the row of search index. If search index is low index of the row, shift the row of search index also. */
420  if ((IDIndex % 2) == 0) {
421  shiftSTDEntryDown((uint32_t *) &pCANAFRam->MASK[StartRow + IDIndex / 2], EntryCnt - IDIndex);
422  }
423  else {
424  shiftSTDEntryDown((uint32_t *) &pCANAFRam->MASK[StartRow + IDIndex / 2 + 1], EntryCnt - IDIndex - 1);
425  }
426 
427  /* Insert new item */
428  tmp = createStdIDEntry(pEntry, IsFullCANEntry);
429  if ((IDIndex % 2) == 0) {
430  if (IDIndex == EntryCnt) {
431  /* Insert unused item if the new item is the last item*/
432  pCANAFRam->MASK[StartRow + IDIndex / 2] = (tmp << 16) | createUnUsedSTDEntry(pEntry->CtrlNo);
433  }
434  else {
435  uint32_t val;
436  val = pCANAFRam->MASK[StartRow + IDIndex / 2] & 0x0000FFFF;
437  /* Insert new item */
438  pCANAFRam->MASK[StartRow + IDIndex / 2] = val | (tmp << 16);
439  }
440  }
441  else {
442  uint32_t val, valNext;
443  val = pCANAFRam->MASK[StartRow + IDIndex / 2];
444  valNext = pCANAFRam->MASK[StartRow + IDIndex / 2 + 1];
445  /* In case the new entry is not the last item, shift the item at the found index to the next row*/
446  if (IDIndex < EntryCnt ) {
447 
448  pCANAFRam->MASK[StartRow + IDIndex / 2 + 1] = (valNext & 0x0000FFFF) | ((val & 0xFFFF) << 16);
449  }
450  pCANAFRam->MASK[StartRow + IDIndex / 2] = (val & 0xFFFF0000) | tmp;
451  }
452 
453  /* Return to previous mode */
454  Chip_CAN_SetAFMode(pCANAF, CurMode);
455  return SUCCESS;
456 }

665  Chip_CAN_SetAFMode(pCANAF, CurMode);

785 /* Set CAN Bit Rate */
786 Status Chip_CAN_SetBitRate(LPC_CAN_T *pCAN, uint32_t BitRate)
787 {
788  IP_CAN_BUS_TIMING_T BusTiming;
789  uint32_t result = 0;
790  uint8_t NT, TSEG1 = 0, TSEG2 = 0;
791  uint32_t CANPclk = 0;
792  uint32_t BRP = 0;
793 
794 #if defined(CHIP_LPC175X_6X)
795  CANPclk = Chip_Clock_GetPeripheralClockRate(Chip_CAN_GetClkIndex(pCAN));
796 #else
797  CANPclk = Chip_Clock_GetPeripheralClockRate();
798 #endif
799  result = CANPclk / BitRate;
800 
801  /* Calculate suitable nominal time value
802  * NT (nominal time) = (TSEG1 + TSEG2 + 3)
803  * NT <= 24
804  * TSEG1 >= 2*TSEG2
805  */
806  for (NT = 24; NT > 0; NT = NT - 2) {
807  if ((result % NT) == 0) {
808  BRP = result / NT - 1;
809 
810  NT--;
811 
812  TSEG2 = (NT / 3) - 1;
813 
814  TSEG1 = NT - (NT / 3) - 1;
815 
816  break;
817  }
818  }
819  if (NT == 0) {
820  return ERROR;
821  }
822 
823  BusTiming.TESG1 = TSEG1;
824  BusTiming.TESG2 = TSEG2;
825  BusTiming.BRP = BRP;
826  BusTiming.SJW = 3;
827  BusTiming.SAM = 0;
828  setBusTiming(pCAN, &BusTiming);
829  return SUCCESS;
830 }

1066 Status Chip_CAN_InsertGroupSTDEntry(LPC_CANAF_T *pCANAF,
1067  LPC_CANAF_RAM_T *pCANAFRam,
1068  CAN_STD_ID_RANGE_ENTRY_T *pEntry)
1069 {
1070  uint16_t InsertIndex = 0;
1071  uint16_t StartRow, EndRow;
1072  uint16_t LowerID = 0, UpperID = 0;
1073  uint16_t i = 0;
1074  CAN_AF_MODE_T CurMode = Chip_CAN_GetAFMode(pCANAF);
1075 
1076  if (getTotalEntryNum(pCANAF) >= CANAF_RAM_ENTRY_NUM) {
1077  return ERROR;
1078  }
1079 
1080  getSectionAddress(pCANAF, CANAF_RAM_SFF_GRP_SEC, &StartRow, &EndRow);
1081 
1082  /* Search for Index of the entry which upper the new item */
1083  for (InsertIndex = StartRow; InsertIndex <= EndRow; InsertIndex++) {
1084  LowerID = (pCANAFRam->MASK[InsertIndex] >> (16 + CAN_STD_ENTRY_ID_POS)) & CAN_STD_ENTRY_ID_MASK;
1085  UpperID = (pCANAFRam->MASK[InsertIndex] >> CAN_STD_ENTRY_ID_POS) & CAN_STD_ENTRY_ID_MASK;
1086  if (LowerID >= pEntry->LowerID.ID_11) {
1087  break;
1088  }
1089  }
1090 
1091  /* Compare to the previous row (if any)*/
1092  if (InsertIndex > 0) {
1093  uint16_t PrevUpperID;
1094  PrevUpperID = (pCANAFRam->MASK[InsertIndex - 1] >> CAN_STD_ENTRY_ID_POS) & CAN_STD_ENTRY_ID_MASK;
1095 
1096  if (PrevUpperID >= pEntry->UpperID.ID_11) {
1097  return SUCCESS;
1098  }
1099 
1100  if (pEntry->UpperID.ID_11 < LowerID) {
1101  if (pEntry->LowerID.ID_11 < PrevUpperID) { /* The new range is merged to the range of the previous row */
1102  uint32_t val = pCANAFRam->MASK[InsertIndex - 1] & 0xFFFF0000;
1103  pCANAFRam->MASK[InsertIndex - 1] = val | createStdIDEntry(&pEntry->UpperID, false);
1104  return SUCCESS;
1105  }
1106  else {
1107  goto insert_grp_entry;
1108  }
1109  }
1110  }
1111 
1112  /* Compare to the next row (if any)*/
1113  if ((EndRow) && (InsertIndex <= EndRow)) {
1114  if (pEntry->UpperID.ID_11 >= UpperID) { /* The new range is merged to the range of the next row */
1115  uint32_t val;
1116  val = createStdIDEntry(&pEntry->LowerID, false) << 16;
1117  val |= createStdIDEntry(&pEntry->UpperID, false);
1118  pCANAFRam->MASK[InsertIndex] = val;
1119  return SUCCESS;
1120  }
1121  else if (pEntry->UpperID.ID_11 < UpperID) {
1122  if (pEntry->UpperID.ID_11 > LowerID) { /* The new range is merged to the range of the next row */
1123  uint32_t val = pCANAFRam->MASK[InsertIndex] & 0x0000FFFF;
1124  pCANAFRam->MASK[InsertIndex] = val | (createStdIDEntry(&pEntry->LowerID, false) << 16);
1125  return SUCCESS;
1126  }
1127  }
1128  }
1129 
1130 insert_grp_entry:
1131 
1132  /* AF Off */
1133  Chip_CAN_SetAFMode(pCANAF, CAN_AF_OFF_MODE);
1134 
1135  /* Move all remaining sections one place down
1136  if new entry will increase FullCAN list */
1137 
1138  for (i = getTotalEntryNum(pCANAF); i > InsertIndex; i--) {
1139  pCANAFRam->MASK[i] = pCANAFRam->MASK[i - 1];
1140  }
1141 
1142  /* Insert new item */
1143  pCANAFRam->MASK[InsertIndex] = createStdIDEntry(&pEntry->LowerID, false) << 16;
1144  pCANAFRam->MASK[InsertIndex] |= createStdIDEntry(&pEntry->UpperID, false);
1145 
1146  getSectionAddress(pCANAF, CANAF_RAM_SFF_GRP_SEC, &StartRow, &EndRow);
1147  setSectionEndAddress(pCANAF, CANAF_RAM_SFF_GRP_SEC, EndRow + 2);
1148  getSectionAddress(pCANAF, CANAF_RAM_EFF_SEC, &StartRow, &EndRow);
