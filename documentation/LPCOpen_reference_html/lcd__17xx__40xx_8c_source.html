54 /* Initialize the LCD controller */
55 void Chip_LCD_Init(LPC_LCD_T *pLCD, LCD_CONFIG_T *LCD_ConfigStruct)
56 {
57  uint32_t i, regValue, *pPal;
58  uint32_t pcd;
59 
60  /* Enable LCD Clock */
61  Chip_Clock_EnablePeriphClock(SYSCTL_CLOCK_LCD);
62 
63  /* disable the display */
64  pLCD->CTRL &= ~CLCDC_LCDCTRL_ENABLE;
65 
66  /* Setting LCD_TIMH register */
67  regValue = ( ((((LCD_ConfigStruct->PPL / 16) - 1) & 0x3F) << 2)
68  | (( (LCD_ConfigStruct->HSW - 1) & 0xFF) << 8)
69  | (( (LCD_ConfigStruct->HFP - 1) & 0xFF) << 16)
70  | (( (LCD_ConfigStruct->HBP - 1) & 0xFF) << 24) );
71  pLCD->TIMH = regValue;
72 
73  /* Setting LCD_TIMV register */
74  regValue = ((((LCD_ConfigStruct->LPP - 1) & 0x3FF) << 0)
75  | (((LCD_ConfigStruct->VSW - 1) & 0x03F) << 10)
76  | (((LCD_ConfigStruct->VFP - 1) & 0x0FF) << 16)
77  | (((LCD_ConfigStruct->VBP - 1) & 0x0FF) << 24) );
78  pLCD->TIMV = regValue;
79 
80  /* Generate the clock and signal polarity control word */
81  regValue = 0;
82  regValue = (((LCD_ConfigStruct->ACB - 1) & 0x1F) << 6);
83  regValue |= (LCD_ConfigStruct->IOE & 1) << 14;
84  regValue |= (LCD_ConfigStruct->IPC & 1) << 13;
85  regValue |= (LCD_ConfigStruct->IHS & 1) << 12;
86  regValue |= (LCD_ConfigStruct->IVS & 1) << 11;
87 
88  /* Compute clocks per line based on panel type */
89  switch (LCD_ConfigStruct->LCD) {
90  case LCD_MONO_4:
91  regValue |= ((((LCD_ConfigStruct->PPL / 4) - 1) & 0x3FF) << 16);
92  break;
93 
94  case LCD_MONO_8:
95  regValue |= ((((LCD_ConfigStruct->PPL / 8) - 1) & 0x3FF) << 16);
96  break;
97 
98  case LCD_CSTN:
99  regValue |= (((((LCD_ConfigStruct->PPL * 3) / 8) - 1) & 0x3FF) << 16);
100  break;
101 
102  case LCD_TFT:
103  default:
104  regValue |= /*1<<26 |*/ (((LCD_ConfigStruct->PPL - 1) & 0x3FF) << 16);
105  }
106 
107  /* panel clock divisor */
108  pcd = 5;// LCD_ConfigStruct->pcd; /* TODO: should be calculated from LCDDCLK */
109  pcd &= 0x3FF;
110  regValue |= ((pcd >> 5) << 27) | ((pcd) & 0x1F);
111  pLCD->POL = regValue;
112 
113  /* disable interrupts */
114  pLCD->INTMSK = 0;
115 
116  /* set bits per pixel */
117  regValue = LCD_ConfigStruct->BPP << 1;
118 
119  /* set color format RGB */
120  regValue |= LCD_ConfigStruct->color_format << 8;
121  regValue |= LCD_ConfigStruct->LCD << 4;
122  if (LCD_ConfigStruct->Dual == 1) {
123  regValue |= 1 << 7;
124  }
125  pLCD->CTRL = regValue;
126 
127  /* clear palette */
128  pPal = (uint32_t *) (&(pLCD->PAL));
129  for (i = 0; i < 128; i++) {
130  *pPal = 0;
131  pPal++;
132  }
133 }

148 /* Write Cursor Image into Internal Cursor Image Buffer */
149 void Chip_LCD_Cursor_WriteImage(LPC_LCD_T *pLCD, uint8_t cursor_num, void *Image)
150 {
151  int i, j;
152  uint32_t *fifoptr, *crsr_ptr = (uint32_t *) Image;
153 
154  /* Check if Cursor Size was configured as 32x32 or 64x64*/
155  if (LCD_Cursor_Size == LCD_CURSOR_32x32) {
156  i = cursor_num * 64;
157  j = i + 64;
158  }
159  else {
160  i = 0;
161  j = 256;
162  }
163  fifoptr = (void *) &(pLCD->CRSR_IMG[0]);
164 
165  /* Copy Cursor Image content to FIFO */
166  for (; i < j; i++) {
167 
168  *fifoptr = *crsr_ptr;
169  crsr_ptr++;
170  fifoptr++;
171  }
172 }

174 /* Load LCD Palette */
175 void Chip_LCD_LoadPalette(LPC_LCD_T *pLCD, void *palette)
176 {
177  LCD_PALETTE_ENTRY_T pal_entry, *ptr_pal_entry;
178  uint8_t i, *pal_ptr;
179  /* This function supports loading of the color palette from
180  the C file generated by the bmp2c utility. It expects the
181  palette to be passed as an array of 32-bit BGR entries having
182  the following format:
183  2:0 - Not used
184  7:3 - Blue
185  10:8 - Not used
186  15:11 - Green
187  18:16 - Not used
188  23:19 - Red
189  31:24 - Not used
190  arg = pointer to input palette table address */
191  ptr_pal_entry = &pal_entry;
192  pal_ptr = (uint8_t *) palette;
193 
194  /* 256 entry in the palette table */
195  for (i = 0; i < 256 / 2; i++) {
196  pal_entry.Bl = (*pal_ptr++) >> 3; /* blue first */
197  pal_entry.Gl = (*pal_ptr++) >> 3; /* get green */
198  pal_entry.Rl = (*pal_ptr++) >> 3; /* get red */
199  pal_ptr++; /* skip over the unused byte */
200  /* do the most significant halfword of the palette */
201  pal_entry.Bu = (*pal_ptr++) >> 3; /* blue first */
202  pal_entry.Gu = (*pal_ptr++) >> 3; /* get green */
203  pal_entry.Ru = (*pal_ptr++) >> 3; /* get red */
204  pal_ptr++; /* skip over the unused byte */
205 
206  pLCD->PAL[i] = *((uint32_t *) ptr_pal_entry);
207  }
208 }
