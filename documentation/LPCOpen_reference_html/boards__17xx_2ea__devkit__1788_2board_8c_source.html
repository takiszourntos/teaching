260 /* Send/Receive data to/from TSC2046. */
261 static void readWriteTSC2046(uint8_t command, uint16_t *data)
262 {
263  uint8_t rx_data[2], tx_data[1] = {0x00};
264 
265  tx_data[0] = command;
266 
267  TSC_CS_ON;
268 
269  Chip_SSP_WriteFrames_Blocking(LCD_SSP_CTRL, tx_data, 1);
270  Chip_SSP_ReadFrames_Blocking(LCD_SSP_CTRL, rx_data, 2);
271 
272 #if (TSC2046_CONVERSION_BITS == 8)
273  *data = (((rx_data[0] << 8) | (rx_data[1])) >> 7) & 0xFF;
274 #else
275  *data = (((rx_data[0] << 8) | rx_data[1]) >> 3) & 0xFFF;
276 #endif
277 
278  TSC_CS_OFF;
279 }

281 /* Evaluate the coordinates received from TSC2046. */
282 static Status evalTSC2046Coord(uint8_t command, uint16_t *coord)
283 {
284  uint32_t i;
285  uint16_t Tmp = 0, previousTmp;
286  int16_t diff = 0;
287  *coord = 0;
288  for (i = 0; i < COORD_GET_NUM; i++) {
289  previousTmp = Tmp;
290  readWriteTSC2046(command, &Tmp);
291  if (Tmp > TSC2046_COORD_MAX) {
292  return ERROR;
293  }
294  if (i > 0) {
295  diff = Tmp - previousTmp;
296  }
297  if (diff < 0) {
298  diff = 0 - diff;
299  }
300  if (diff > TSC2046_DELTA_VARIANCE) {
301  return ERROR;
302  }
303  *coord += Tmp;
304  }
305  *coord /= COORD_GET_NUM;
306  return SUCCESS;
307 }

309 /* Convert the coord received from TSC to a value on truly LCD */
310 static int16_t calibrateTSC2046(int16_t Coord, int16_t MinVal, int16_t MaxVal, int16_t TrueSize)
311 {
312  int16_t tmp;
313  int16_t ret;
314  uint8_t convert = 0;
315 
316  /* Swap value? */
317  if (MinVal > MaxVal) {
318  tmp = MaxVal;
319  MaxVal = MinVal;
320  MinVal = tmp;
321  convert = 1;
322  }
323 
324  ret = (Coord - MinVal) * TrueSize / (MaxVal - MinVal);
325  if (convert) {
326  ret = TrueSize - ret;
327  }
328 
329  return ret;
330 }

596 /* Get Touch coordinates */
597 bool Board_GetTouchPos(int16_t *pX, int16_t *pY)
598 {
599  uint16_t tmp;
600  uint16_t x, y, z1, z2, z = 0;
601  Status Sts = SUCCESS;
602 
603  readWriteTSC2046(X_MEASURE, &x);
604  readWriteTSC2046(Y_MEASURE, &y);
605  readWriteTSC2046(Z1_MEASURE, &z1);
606  readWriteTSC2046(Z2_MEASURE, &z2);
607 
608  if (z1 != 0) {
609  z = x * ((z2 / z1) - 1);
610  }
611  if ((z <= 0) || (z > 35000)) {
612  return false;
613  }
614  /* Get X-Coordinate */
615  Sts = evalTSC2046Coord(X_MEASURE, &x);
616 
617  if (Sts == ERROR) {
618  return false;
619  }
620  /* Get Y-Coordinate */
621  Sts = evalTSC2046Coord(Y_MEASURE, &y);
622  if (Sts == ERROR) {
623  return false;
624  }
625  /* Get Z1-Value */
626  Sts = evalTSC2046Coord(Z1_MEASURE, &z1);
627  if (Sts == ERROR) {
628  return false;
629  }
630  /* Get Z2-Value */
631  Sts = evalTSC2046Coord(Z2_MEASURE, &z2);
632  if (Sts == ERROR) {
633  return false;
634  }
635 
636  z = x * ((z2 / z1) - 1);
637  if ((z <= 0) || (z > 35000)) {
638  return false;
639  }
640  else {
641  /* Swap, adjust to truly size of LCD */
642  if (TSC_Config.swap_xy) {
643  *pY = calibrateTSC2046(x, TSC_Config.ad_top, TSC_Config.ad_bottom, TSC_Config.lcd_height);
644  *pX = calibrateTSC2046(y, TSC_Config.ad_left, TSC_Config.ad_right, TSC_Config.lcd_width);
645  }
646  else {
647  *pX = calibrateTSC2046(x, TSC_Config.ad_top, TSC_Config.ad_bottom, TSC_Config.lcd_width);
648  *pY = calibrateTSC2046(y, TSC_Config.ad_left, TSC_Config.ad_right, TSC_Config.lcd_height);
649  }
650  }
651  readWriteTSC2046(PWRDOWN, &tmp);
652 
653  return true;
654 }
