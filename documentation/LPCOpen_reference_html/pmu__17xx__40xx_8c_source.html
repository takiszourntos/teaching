50 /* Enter MCU Sleep mode */
51 void Chip_PMU_SleepState(LPC_PMU_T *pPMU)
52 {
53  /* Select Sleep/Deep Sleep mode */
54  pPMU->PCON &= ~(PMU_PCON_PM1_FLAG | PMU_PCON_PM0_FLAG);
55 
56  /* Clearing SLEEPDEEP bit in SCR makes it Sleep mode */
57  SCB->SCR &= ~(1UL << SCB_SCR_SLEEPDEEP_Pos);
58 
59  /* Enter sleep mode */
60  __WFI();
61 }

76 /* Enter MCU Power down mode */
77 void Chip_PMU_PowerDownState(LPC_PMU_T *pPMU)
78 {
79  /* Select power down mode */
80  pPMU->PCON = (pPMU->PCON & ~PMU_PCON_PM1_FLAG) | PMU_PCON_PM0_FLAG;
81 
82  /* Setting SLEEPDEEP bit in SCR makes it power down mode */
83  SCB->SCR |= (1UL << SCB_SCR_SLEEPDEEP_Pos);
84 
85  /* Enter sleep mode */
86  __WFI();
87 }

102 /* Put some of the peripheral in sleep mode */
103 void Chip_PMU_Sleep(LPC_PMU_T *pPMU, CHIP_PMU_MCUPOWER_T SleepMode)
104 {
105  if (SleepMode == PMU_MCU_DEEP_SLEEP) {
106  Chip_PMU_DeepSleepState(pPMU);
107  }
108  else if (SleepMode == PMU_MCU_POWER_DOWN) {
109  Chip_PMU_PowerDownState(pPMU);
110  }
111  else if (SleepMode == PMU_MCU_DEEP_PWRDOWN) {
112  Chip_PMU_DeepPowerDownState(pPMU);
113  }
114  else {
115  /* PMU_MCU_SLEEP */
116  Chip_PMU_SleepState(pPMU);
117  }
