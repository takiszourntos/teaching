214 /* Initialize pin muxing for SPI interface */
215 void Board_SPI_Init(bool isMaster)
216 {
217  /* Set up clock and muxing for SSP0 interface */
218  /*
219  * Initialize SSP0 pins connect
220  * P0.15: SCK
221  * P0.16: SSEL
222  * P0.17: MISO
223  * P0.18: MOSI
224  */
225  Chip_IOCON_PinMux(LPC_IOCON, 0, 15, IOCON_MODE_PULLDOWN, IOCON_FUNC3);
226  if (isMaster) {
227  Chip_IOCON_PinMux(LPC_IOCON, 0, 16, IOCON_MODE_PULLUP, IOCON_FUNC0);
228  Chip_GPIO_WriteDirBit(LPC_GPIO, 0, 16, true);
229  Board_SPI_DeassertSSEL();
230 
231  }
232  else {
233  Chip_IOCON_PinMux(LPC_IOCON, 0, 16, IOCON_MODE_PULLUP, IOCON_FUNC3);

235  Chip_IOCON_PinMux(LPC_IOCON, 0, 17, IOCON_MODE_INACT, IOCON_FUNC3);
236  Chip_IOCON_PinMux(LPC_IOCON, 0, 18, IOCON_MODE_INACT, IOCON_FUNC3);

264 /* Sets up board specific I2C interface */
265 void Board_I2C_Init(I2C_ID_T id)
266 {
267  switch (id) {
268  case I2C0:
269  Chip_IOCON_PinMux(LPC_IOCON, 0, 27, IOCON_MODE_INACT, IOCON_FUNC1);
270  Chip_IOCON_PinMux(LPC_IOCON, 0, 28, IOCON_MODE_INACT, IOCON_FUNC1);
271  Chip_IOCON_SetI2CPad(LPC_IOCON, I2CPADCFG_STD_MODE);
272  break;
273 
274  case I2C1:
275  Chip_IOCON_PinMux(LPC_IOCON, 0, 19, IOCON_MODE_INACT, IOCON_FUNC2);
276  Chip_IOCON_PinMux(LPC_IOCON, 0, 20, IOCON_MODE_INACT, IOCON_FUNC2);
277  Chip_IOCON_EnableOD(LPC_IOCON, 0, 19);
278  Chip_IOCON_EnableOD(LPC_IOCON, 0, 20);
279  break;
280 
281  case I2C2:
282  Chip_IOCON_PinMux(LPC_IOCON, 0, 10, IOCON_MODE_INACT, IOCON_FUNC2);
283  Chip_IOCON_PinMux(LPC_IOCON, 0, 11, IOCON_MODE_INACT, IOCON_FUNC2);
284  Chip_IOCON_EnableOD(LPC_IOCON, 0, 10);
285  Chip_IOCON_EnableOD(LPC_IOCON, 0, 11);
286  break;
287  }
288 }
