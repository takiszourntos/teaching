188 /* Setup system clocking */
189 void Board_SetupClocking(void)
190 {
191  /* Enable PBOOST for CPU clock over 100MHz */
192  Chip_SYSCTL_EnableBoost();
193 
194  Chip_SetupXtalClocking();
195 
196  /* SPIFI clocking will be derived from Main PLL with a divider of 2 (60MHz) */
197  Chip_Clock_SetSPIFIClockDiv(2);
198  Chip_Clock_SetSPIFIClockSource(SYSCTL_SPIFICLKSRC_MAINPLL);
199 
200 }

202 /* Sets up system pin muxing */
203 void Board_SetupMuxing(void)
204 {
205  int i, j;
206 
207  Chip_Clock_EnablePeriphClock(SYSCTL_CLOCK_EMC);
208  Chip_SYSCTL_PeriphReset(SYSCTL_RESET_IOCON);
209 
210  /* Setup data, address, and EMC control pins with high slew rate */
211  for (i = 3; i <= 4; i++) {
212  for (j = 0; j <= 31; j++) {
213  Chip_IOCON_PinMuxSet(LPC_IOCON, (uint8_t) i, (uint8_t) j, (IOCON_FUNC1 | IOCON_FASTSLEW_EN));
214  }
215  }
216  for (i = 16; i <= 31; i++) {
217  Chip_IOCON_PinMuxSet(LPC_IOCON, 2, (uint8_t) i, (IOCON_FUNC1 | IOCON_FASTSLEW_EN));
218  }
219 
220  /* Setup system level pin muxing */
221  Chip_IOCON_SetPinMuxing(LPC_IOCON, pinmuxing, sizeof(pinmuxing) / sizeof(PINMUX_GRP_T));
222 }

224 /* Setup external memories */
225 void Board_SetupExtMemory(void)
226 {
227  /* Setup EMC Delays */
228  /* Move all clock delays together */
229  LPC_SYSCTL->EMCDLYCTL = (CLK0_DELAY) | (CLK0_DELAY << 8) | (CLK0_DELAY << 16 | (CLK0_DELAY << 24));
230 
231  /* Setup EMC Clock Divider for divide by 2 */
232  /* Setup EMC clock for a divider of 2 from CPU clock. Enable EMC clock for
233  external memory setup of DRAM. */
234  Chip_Clock_SetEMCClockDiv(SYSCTL_EMC_DIV2);
235  Chip_SYSCTL_PeriphReset(SYSCTL_RESET_EMC);
236 
237  /* Init EMC Controller -Enable-LE mode- clock ratio 1:1 */
238  Chip_EMC_Init(1, 0, 0);
239 
240  /* Init EMC Dynamic Controller */
241  Chip_EMC_Dynamic_Init((IP_EMC_DYN_CONFIG_T *) &IS42S32800D_config);
242 
243  /* Init EMC Static Controller CS0 */
244  Chip_EMC_Static_Init((IP_EMC_STATIC_CONFIG_T *) &SST39VF320_config);
245 
246 
247  /* Init EMC Static Controller CS1 */
248  Chip_EMC_Static_Init((IP_EMC_STATIC_CONFIG_T *) &K9F1G_config);
249 
250  /* EMC Shift Control */
251  LPC_SYSCTL->SCS |= 1;
252 }
