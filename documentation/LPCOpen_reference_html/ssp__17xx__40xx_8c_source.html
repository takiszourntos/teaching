147 void Chip_SSP_SetClockRate(LPC_SSP_T *pSSP, uint32_t clk_rate, uint32_t prescale)
148 {
149  uint32_t temp;
150  temp = pSSP->CR0 & (~(SSP_CR0_SCR(0xFF)));

155 /* SSP Polling Read/Write in blocking mode */
156 uint32_t Chip_SSP_RWFrames_Blocking(LPC_SSP_T *pSSP, Chip_SSP_DATA_SETUP_T *xf_setup)
157 {
158  /* Clear all remaining frames in RX FIFO */
159  while (Chip_SSP_GetStatus(pSSP, SSP_STAT_RNE)) {
160  Chip_SSP_ReceiveFrame(pSSP);
161  }
162 
163  /* Clear status */
164  Chip_SSP_ClearIntPending(pSSP, SSP_INT_CLEAR_BITMASK);
165 
166  if (Chip_SSP_GetDataSize(pSSP) > SSP_BITS_8) {
167  while (xf_setup->rx_cnt < xf_setup->length || xf_setup->tx_cnt < xf_setup->length) {
168  /* write data to buffer */
169  if (( Chip_SSP_GetStatus(pSSP, SSP_STAT_TNF) == SET) && ( xf_setup->tx_cnt < xf_setup->length) ) {
170  SSP_Write2BFifo(pSSP, xf_setup);
171  }
172 
173  /* Check overrun error */
174  if (Chip_SSP_GetRawIntStatus(pSSP, SSP_RORRIS) == SET) {
175  return ERROR;
176  }
177 
178  /* Check for any data available in RX FIFO */
179  SSP_Read2BFifo(pSSP, xf_setup);
180  }
181  }
182  else {
183  while (xf_setup->rx_cnt < xf_setup->length || xf_setup->tx_cnt < xf_setup->length) {
184  /* write data to buffer */
185  if (( Chip_SSP_GetStatus(pSSP, SSP_STAT_TNF) == SET) && ( xf_setup->tx_cnt < xf_setup->length) ) {
186  SSP_Write1BFifo(pSSP, xf_setup);
187  }
188 
189  /* Check overrun error */
190  if (Chip_SSP_GetRawIntStatus(pSSP, SSP_RORRIS) == SET) {
191  return ERROR;
192  }
193 
194  /* Check for any data available in RX FIFO */
195  SSP_Read1BFifo(pSSP, xf_setup);
196  }
197  }
198  if (xf_setup->tx_data) {
199  return xf_setup->tx_cnt;
200  }
201  else if (xf_setup->rx_data) {
202  return xf_setup->rx_cnt;
203  }
204 
205  return 0;
206 }

361 Status Chip_SSP_Int_RWFrames8Bits(LPC_SSP_T *pSSP, Chip_SSP_DATA_SETUP_T *xf_setup)
362 {
363  /* Check overrun error in RIS register */
364  if (Chip_SSP_GetRawIntStatus(pSSP, SSP_RORRIS) == SET) {
365  return ERROR;
366  }
367 
368  if ((xf_setup->tx_cnt != xf_setup->length) || (xf_setup->rx_cnt != xf_setup->length)) {
369  /* check if RX FIFO contains data */
370  SSP_Read1BFifo(pSSP, xf_setup);
371 
372  while ((Chip_SSP_GetStatus(pSSP, SSP_STAT_TNF)) && (xf_setup->tx_cnt != xf_setup->length)) {
373  /* Write data to buffer */

375 
376  /* Check overrun error in RIS register */

378  return ERROR;

380

382  SSP_Read1BFifo(pSSP, xf_setup);

433 /* Set the clock frequency for SSP interface */
434 void Chip_SSP_SetBitRate(LPC_SSP_T *pSSP, uint32_t bitRate)
435 {
436  uint32_t ssp_clk, cr0_div, cmp_clk, prescale;
437 
438 #if defined(CHIP_LPC175X_6X)
439  ssp_clk = Chip_Clock_GetPeripheralClockRate(Chip_SSP_GetClkIndex(pSSP));
440 #else
441  ssp_clk = Chip_Clock_GetPeripheralClockRate();
442 #endif
443 
444  cr0_div = 0;
445  cmp_clk = 0xFFFFFFFF;
446  prescale = 2;
447 
448  while (cmp_clk > bitRate) {
449  cmp_clk = ssp_clk / ((cr0_div + 1) * prescale);
450  if (cmp_clk > bitRate) {
451  cr0_div++;
452  if (cr0_div > 0xFF) {
453  cr0_div = 0;
454  prescale += 2;
455  }
456  }
457  }
458 
459  Chip_SSP_SetClockRate(pSSP, cr0_div, prescale);
460 }
