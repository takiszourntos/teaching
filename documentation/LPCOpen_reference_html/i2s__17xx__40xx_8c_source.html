46 /* Get divider value */
47 STATIC Status getClkDiv(LPC_I2S_T *pI2S, I2S_AUDIO_FORMAT_T *format, uint16_t *pxDiv, uint16_t *pyDiv, uint32_t *pN)
48 {
49  uint32_t pClk;
50  uint32_t x, y;
51  uint64_t divider;
52  uint16_t dif;
53  uint16_t xDiv = 0, yDiv = 0;
54  uint32_t N;
55  uint16_t err, ErrorOptimal = 0xFFFF;
56 
57 #if defined(CHIP_LPC175X_6X)
58  pClk = Chip_Clock_GetPeripheralClockRate(SYSCTL_PCLK_I2S);
59 #else
60  pClk = Chip_Clock_GetPeripheralClockRate();
61 #endif
62 
63  /* divider is a fixed point number with 16 fractional bits */
64  divider = (((uint64_t) (format->SampleRate) * 2 * (format->WordWidth) * 2) << 16) / pClk;
65  /* find N that make x/y <= 1 -> divider <= 2^16 */
66  for (N = 64; N > 0; N--) {
67  if ((divider * N) < (1 << 16)) {
68  break;
69  }
70  }
71  if (N == 0) {
72  return ERROR;
73  }
74  divider *= N;
75  for (y = 255; y > 0; y--) {
76  x = y * divider;
77  if (x & (0xFF000000)) {
78  continue;
79  }
80  dif = x & 0xFFFF;
81  if (dif > 0x8000) {
82  err = 0x10000 - dif;
83  }
84  else {
85  err = dif;
86  }
87  if (err == 0) {
88  yDiv = y;
89  break;
90  }
91  else if (err < ErrorOptimal) {
92  ErrorOptimal = err;
93  yDiv = y;
94  }
95  }
96  xDiv = ((uint64_t) yDiv * (format->SampleRate) * 2 * (format->WordWidth) * N * 2) / pClk;
97  if (xDiv >= 256) {
98  xDiv = 0xFF;
99  }
100  if (xDiv == 0) {
101  xDiv = 1;
102  }
103 
104  *pxDiv = xDiv;
105  *pyDiv = yDiv;
106  *pN = N;
107  return SUCCESS;
108 }

133 /* Configure I2S for Audio Format input */
134 Status Chip_I2S_TxConfig(LPC_I2S_T *pI2S, I2S_AUDIO_FORMAT_T *format)
135 {
136  uint32_t temp;
137  uint16_t xDiv, yDiv;
138  uint32_t N;
139 
140  if (getClkDiv(pI2S, format, &xDiv, &yDiv, &N) == ERROR) {
141  return ERROR;
142  }
143 
144  temp = pI2S->DAO & (~(I2S_DAO_WORDWIDTH_MASK | I2S_DAO_MONO | I2S_DAO_SLAVE | I2S_DAO_WS_HALFPERIOD_MASK));
145  if (format->WordWidth <= 8) {
146  temp |= I2S_WORDWIDTH_8;
147  }
148  else if (format->WordWidth <= 16) {
149  temp |= I2S_WORDWIDTH_16;
150  }
151  else {
152  temp |= I2S_WORDWIDTH_32;
153  }
154 
155  temp |= (format->ChannelNumber) == 1 ? I2S_MONO : I2S_STEREO;
156  temp |= I2S_MASTER_MODE;
157  temp |= I2S_DAO_WS_HALFPERIOD(format->WordWidth - 1);
158  pI2S->DAO = temp;
159  pI2S->TXMODE = I2S_TXMODE_CLKSEL(0);
160  pI2S->TXBITRATE = N - 1;
161  pI2S->TXRATE = yDiv | (xDiv << 8);
162  return SUCCESS;
163 }
