49 #if defined(CHIP_LPC175X_6X)
50 void Chip_SetupIrcClocking(void)
51 {
52  /* Disconnect the Main PLL if it is connected already */
53  if (Chip_Clock_IsMainPLLConnected()) {
54  Chip_Clock_DisablePLL(SYSCTL_MAIN_PLL, SYSCTL_PLL_CONNECT);
55  }
56 
57  /* Disable the PLL if it is enabled */
58  if (Chip_Clock_IsMainPLLEnabled()) {
59  Chip_Clock_DisablePLL(SYSCTL_MAIN_PLL, SYSCTL_PLL_ENABLE);
60  }
61 
62  Chip_Clock_SetCPUClockDiv(0);
63  Chip_Clock_SetMainPLLSource(SYSCTL_PLLCLKSRC_IRC);
64 
65  /* FCCO = ((44+1) * 2 * 4MHz) / (0+1) = 360MHz */
66  Chip_Clock_SetupPLL(SYSCTL_MAIN_PLL, 44, 0);
67 
68  Chip_Clock_EnablePLL(SYSCTL_MAIN_PLL, SYSCTL_PLL_ENABLE);
69 
70  Chip_Clock_SetCPUClockDiv(2);
71  while (!Chip_Clock_IsMainPLLLocked()) {} /* Wait for the PLL to Lock */
72 
73  Chip_Clock_EnablePLL(SYSCTL_MAIN_PLL, SYSCTL_PLL_CONNECT);
74 }

111 /* Clock and PLL initialization based on the internal oscillator */
112 void Chip_SetupIrcClocking(void)
113 {
114  /* Clock the CPU from SYSCLK, in case if it is clocked by PLL0 */
115  Chip_Clock_SetCPUClockSource(SYSCTL_CCLKSRC_SYSCLK);
116 
117  /* Disable the PLL if it is enabled */
118  if (Chip_Clock_IsMainPLLEnabled()) {
119  Chip_Clock_DisablePLL(SYSCTL_MAIN_PLL, SYSCTL_PLL_ENABLE);
120  }
121 
122  /* It is safe to switch the PLL Source to IRC */
123  Chip_Clock_SetMainPLLSource(SYSCTL_PLLCLKSRC_IRC);
124 
125  /* FCCO = 12MHz * (9+1) * 2 * (0+1) = 240MHz */
126  /* Fout = FCCO / ((0+1) * 2) = 120MHz */
127  Chip_Clock_SetupPLL(SYSCTL_MAIN_PLL, 9, 0);
128 
129  Chip_Clock_EnablePLL(SYSCTL_MAIN_PLL, SYSCTL_PLL_ENABLE);
130  Chip_Clock_SetCPUClockDiv(1);
131  while (!Chip_Clock_IsMainPLLLocked()) {} /* Wait for the PLL to Lock */
132  Chip_Clock_SetCPUClockSource(SYSCTL_CCLKSRC_MAINPLL);
133 
134  /* Peripheral clocking will be derived from PLL0 with a divider of 2 (60MHz) */
135  Chip_Clock_SetPCLKDiv(2);
136 }
