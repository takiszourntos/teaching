56 /* Initialize stopwatch */
57 void StopWatch_Init(void)
58 {
59  /* Use timer 1. Set prescaler to divide by 8, should give ticks at 3.75 MHz.
60  That gives a useable stopwatch measurement range of about 19 minutes
61  (if system clock is running at 120 MHz). */
62  const uint32_t prescaleDivisor = 8;
63  Chip_TIMER_Init(LPC_TIMER1);
64  Chip_TIMER_PrescaleSet(LPC_TIMER1, prescaleDivisor - 1);
65  Chip_TIMER_Enable(LPC_TIMER1);
66 
67  /* Pre-compute tick rate. Note that peripheral clock supplied to the
68  timer includes a fixed divide by 4. */
69  ticksPerSecond = Chip_Clock_GetSystemClockRate() / prescaleDivisor / 4;
70  ticksPerMs = ticksPerSecond / 1000;
71  ticksPerUs = ticksPerSecond / 1000000;
72 }
